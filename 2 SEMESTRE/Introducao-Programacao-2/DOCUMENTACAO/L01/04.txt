/* A função recebe um inteiro n e um vetor v de n elementos
 * e devolve a posição do menor elemento do vetor v */
int menor_pos(int n, int v[]) 
{
   int i, pos = 0;
   /* v[pos] e o menor elemento de v[0..i-1] */
   for (i = 1; i < n; i++) {
      if (v[i] < v[pos])
         pos = i;      
   }
   return pos;
}
